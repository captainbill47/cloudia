class BrainFiles:
    def __init__(self):
        self.files = {
            "soul": os.path.join(BRAIN_PATH, "SOUL.md"),
            "identity": os.path.join(BRAIN_PATH, "IDENTITY.md"),
            "existence": os.path.join(BRAIN_PATH, "EXISTENCE.md"),

            "projects": os.path.join(BRAIN_PATH, "projects/")
        }
        
    def read_all(self):
        """Load all brain files for context window"""
        context = ""
        for name, path in self.files.items():
            if os.path.isdir(path):
                # Read all project files
                for proj in glob.glob(os.path.join(path, "*.md")):
                    with open(proj, 'r') as f:
                        context += f"\n\n--- {os.path.basename(proj)} ---\n{f.read()}"
            else:
                with open(path, 'r') as f:
                    context += f"\n\n--- {name.upper()} ---\n{f.read()}"
        return context
    
    def write_memory(self, entry):
        """Append to MEMORY.md with timestamp"""
        from datetime import datetime
        timestamp = datetime.utcnow().strftime("%Y-%m-%d %H:%M UTC")
        with open(self.files["memory"], 'a') as f:
            f.write(f"\n\n## {timestamp}\n{entry}\n")
        # Trigger sync
        os.system("/root/.openclaw/workspace/scripts/brain-sync.sh")
